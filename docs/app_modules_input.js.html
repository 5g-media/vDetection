 * @fires Output.start
 * @fires Output.stop

var vDetection = {
	_timer: null,
	start: function (environment) {
		var deferred = Q.defer();
		const timestamp = Date.now();
		Input.onStart(environment)
			.then(function (data) {
				const now = Date.now();
				Output.start(environment, data);
				deferred.resolve({
					timestamp: timestamp,
					elapsed: now - timestamp,
					info: environment.info()
				});
			}).catch(function (error) {
				deferred.reject({
					timestamp: timestamp,
					message: err,
					info: environment.info()
				});
			});
		return deferred.promise;
	},
	stop: function (environment) {

				deferred.resolve({
					timestamp: timestamp,

					info: environment.info()
				});

				deferred.reject({
					timestamp: timestamp,
					message: err,
					info: environment.info()

	}
